
CREATE TABLE STATION(
    ID NUMBER,
    CITY VARCHAR(21),
    STATE VARCHAR(2),
    LAT_N NUMBER,
    LONG_W NUMBER
);

SELECT CITY,STATE FROM STATION
WHERE ID BETWEEN 1 AND 20;

SELECT *
FROM STATION

SELECT DISTINCT ID,CITY FROM STATION
WHERE ID % 2 = 0
ORDER BY ID

SELECT COUNT(*) - COUNT(DISTINCT CITY) AS DIFFERENCE
FROM STATION;

SELECT DISTINCT CITY FROM STATION
WHERE CITY ILIKE 'a%' OR CITY ILIKE 'e%' OR CITY ILIKE 'i%' OR CITY ILIKE 'o%' OR CITY ILIKE 'u%';

SELECT DISTINCT CITY FROM STATION
WHERE CITY ILIKE '%a' OR CITY ILIKE '%e' OR CITY ILIKE '%i' OR CITY ILIKE '%o' OR CITY ILIKE '%u';

SELECT DISTINCT CITY
FROM STATION
WHERE CITY NOT ILIKE 'a%' 
  AND CITY NOT ILIKE 'e%' 
  AND CITY NOT ILIKE 'i%' 
  AND CITY NOT ILIKE 'o%' 
  AND CITY NOT ILIKE 'u%';

SELECT DISTINCT CITY
FROM STATION
WHERE CITY NOT ILIKE '%a' 
  AND CITY NOT ILIKE '%e' 
  AND CITY NOT ILIKE '%i' 
  AND CITY NOT ILIKE '%o' 
  AND CITY NOT ILIKE '%u';

SELECT DISTINCT CITY
FROM STATION
WHERE NOT (CITY ILIKE 'a%' OR CITY ILIKE 'e%' OR CITY ILIKE 'i%' OR CITY ILIKE 'o%' OR CITY ILIKE 'u%')
   AND NOT (CITY ILIKE '%a' OR CITY ILIKE '%e' OR CITY ILIKE '%i' OR CITY ILIKE '%o' OR CITY ILIKE '%u');

SELECT DISTINCT CITY
FROM STATION
WHERE NOT (CITY ILIKE 'a%' OR CITY ILIKE 'e%' OR CITY ILIKE 'i%' OR CITY ILIKE 'o%' OR CITY ILIKE 'u%')
  AND NOT (CITY ILIKE '%a' OR CITY ILIKE '%e' OR CITY ILIKE '%i' OR CITY ILIKE '%o' OR CITY ILIKE '%u');
